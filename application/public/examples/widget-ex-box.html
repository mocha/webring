Widget code

You can copy and paste this code into your own HTML file to add a widget to your site.

<script src="https://webring.fun/widget.js"></script>
<script>
    // Complete configuration of the box style widget
    webring.init({
        // Required - specifies the widget type
        type: "box",
        
        // Optional - join a specific ringlet
        // ringlet: "today-in-tabs",
        
        // Visual appearance
        color: "#e74c3c", // Custom hex color for the widget
        opacity: 0.9, // Opacity between 0.2 and 1.0

        // Content options
        // widget_content: "I'm a member of a webring and it's an extremely cool thing to do with my time.",
        
        // Position options
        position: "bottom-right", // Placement: "top-left", "top-right", "bottom-left", "bottom-right"
        
        // Behavior options
        start_collapsed: false // Whether to start as small badge (true) or full widget (false)
    });
</script>
